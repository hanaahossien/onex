مصنع أونكس لأللمنيوم والواجهات

تــم تأســيس مصنــع إعمــار الجنــوب لأللمنيــوم والواجهــات فــي عــام 2010 م حامــا إســم المجموعــة ليكــون أحــد فروعهــا فــي مجــال أعمــال األلمونيــوم والواجهــات وداعــم أساســي للمجموعــة فــي تنفيــذ مشــاريعها الحكوميــة والخاصــة ، ثــم بعــد ذلــك بنــاء علــى رغبــة االدارة ان يكــون هنالــك مـة بذاتـه ولـه هويتـه الخاصـة وبصمتـه ولكـي يكـون عال مسـمى تجـاري للمصنـع ليكـون مسـتقال متميـزة فـي مجـال االلمنيـوم والواجهـات داخـل المملكـة وخارجهـا، لـذا تقـرر تغييـر المسـمى إلـى مصنــع اونكــس لأللمنيــوم والواجهــات ،

ومنــذ تاريــخ التأســيس نحــن نضــع نصــب اعيننــا تطويــر منتجاتنـا وخدماتنـا واتبـاع اسـتراتيجية واضحـة ومحـددة تهـدف إلـى التحديـث والتطويـر المسـتمر دينـج وتصنيـع وتركيـب فـي مجـال تصميـم وتصنيـع الواجهـات الزجاجيـة وتصميـم وتركيـب الكال األبـواب والشـبابيك ممايتيـح تقديـم الحلـول للمقترحـات األكثـر تعقيـدًا فـي صناعـة األلمنيـوم و دينــج . الزجــاج والكال كمـا يحـرص مصنـع أونكـس لأللمنيـوم والواجهـات وبصـورة متواصلـة علـى إبقـاء عمليـة تطويـر المصنــع مســتمرة وتطبيــق أســاليب ومفاهيــم جديــدة ومبتكــرة فــي قطــاع األلمونيــوم والزجــاج واألبـواب والديكـور تماشـيًا مـع النهضـة العمرانيـة والتنمويـة والتقـدم الكبيـر الذي تشـهده المملكة العربيـة السـعودية وذلـك مـن خـال إتبـاع سياسـة جـودة مدروسـة ومنظمـة والعمـل علـى تلبيـة إحتياجـات السـوق باإلضافـة إلـى تحقيـق رضـاء العمـاء ومتطلباتهـم ورغباتهـم وتوقعاتهـم فـي الحصـول علىأفضـل الخدمـات وبأسـعار تنافسـية .

ئــم مــع متطلبــات النظــم وقــام المصنــع بعمليــة التطويــر والتنفيــذ ألنظمــة العمــل لديــه لتتال اإلداريـة العالميـة إلعطـاء الضمـان الكافـي فـي توجيـة كوادرالمصنـع لكيفيـة أداء عملهـم والتحقـق ومنهجيــة قائمــة علــى التطويــر مــن أن جميــع عمليــات المصنــع تتــم بطريقــة مجديــة إقتصاديــًا قــة والتفكيــر المبنــي والتخطيــط الســليم المبنــي علــى تحديــد العمــاء والمهتميــن وذوي العال ّ علــى المخاطــر والفــرص وذلــك وفقــا لمتطلبــات مواصفــة نظــام إدارة الجــودة مة والصحة المهنية ومتطلبات السال ونحـن اآلن ننظـر إلـى المسـتقبل بعيـن التفـاؤل واألمل ونسـعى لزيـادة فروعنا ومشـاريعنا والحفاظ علـى السـمعة الطيبـة التـي طالمـا سـاهمت فـي وضـع المصنـع فـي منصـب رائـد علـى مسـتوى المنطقـة ممـا يتيـح لـه االسـتجابة التنافسـية والشـاملة لمواجهـة التحديـات التقنيـة فـي المجـال .
